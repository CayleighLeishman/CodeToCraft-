Slideshow CSS:

 /* Slideshow Stuff on Index page */
 * {
     box-sizing: border-box
 }
 
    .mySlides {
       display: none
    }
    .slideshow-container img {
        vertical-align: middle;
    }

    .slideshow-container {
        /* GRID STUFF, dont delete until you're happy with the grid layout of home-main and know you dont need it to span 1/3 */
       /* grid-column: 1/3; */
       max-width: 25rem;
       position: relative;
       margin: auto;

       z-index:2;
    }

    /* Next & previous buttons on Slideshow*/
    .prev, .next {
      cursor: pointer;
      position: absolute;
      top: 50%;
      width: auto;  /* this makes the left arrow go way out of view of page */
      padding: 0;
      margin-top: .2rem;
      /* margin-top: -22px; */
      color: var(--mulan);
      font-weight: bold;
      font-size: 8rem;
      transition: 0.6s ease;
      border-radius: 0 3px 3px 0;
      user-select: none;
    }

    /* Position the "next button" to the right on slideshow */
    .next {
      right: 0;
         border-radius: 3px 0 0 3px;
       }

       /* On hover, add a Who-knows background color with a little bit see-through */
       .prev:hover, .next:hover {
         background-color:var(--who-knows)(0,0,0,0.8);
       }

    /* Caption text */
    .text {
      color: var(--mulan);
      font-size: 15px;
      padding: 8px 12px;
      position: absolute;
      bottom: 8px;
      width: 100%;
      text-align: center;
    }

    /* Number text (1/3 etc) */
    .numbertext {
      color: var(--elsa);
      font-size: 12px;
      padding: 8px 12px;
      position: absolute;
      top: 0;
    }

    /* The dots/bullets/indicators */
    .dot {
      cursor: pointer;
      height: 15px;
      width: 15px;
      margin: 0 2px;
      background-color: var(--cinderella);
      border-radius: 50%;
      display: inline-block;
      transition: background-color 0.6s ease;
    }

    .active, .dot:hover {
      background-color: var(--mulan);
    }

    /* Fading animation */
    .fade {
      animation-name: fade;
      animation-duration: 1.5s;
    }

    @keyframes fade {
      from {opacity: .4} 
      to {opacity: 1}
    }
